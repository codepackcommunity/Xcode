rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Helper function to check if user is authenticated
    function isAuthenticated() {
      return request.auth != null;
    }
    
    // Helper function to check if user is admin
    function isAdmin() {
      return isAuthenticated() && 
        exists(/databases/$(database)/documents/users/$(request.auth.uid)) &&
        get(/databases/$(database)/documents/users/$(request.auth.uid)).data.isAdmin == true;
    }
    
    // Helper function to check if user is sales personnel
    function isSalesPersonnel() {
      return isAuthenticated() && 
        exists(/databases/$(database)/documents/users/$(request.auth.uid)) &&
        (get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == 'sales' ||
         get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == 'dataEntry');
    }
    
    // Helper function to check if user is approved
    function isUserApproved() {
      return isAuthenticated() && 
        exists(/databases/$(database)/documents/users/$(request.auth.uid)) &&
        (get(/databases/$(database)/documents/users/$(request.auth.uid)).data.status == 'approved' ||
         get(/databases/$(database)/documents/users/$(request.auth.uid)).data.status == 'active' ||
         get(/databases/$(database)/documents/users/$(request.auth.uid)).data.isAdmin == true);
    }

    // Users collection
    match /users/{userId} {
      allow read, write: if isAdmin();
      allow create: if isAuthenticated() && request.auth.uid == userId;
      allow read, update: if isAuthenticated() && request.auth.uid == userId;
      allow list: if isAuthenticated();
    }
    
    // Stocks collection - SIMPLIFIED for data retrieval
    match /stocks/{stockId} {
      allow read: if isAuthenticated() && isUserApproved();
      allow create, delete: if isAdmin();
      allow update: if isAdmin() || isSalesPersonnel();
    }
    
    // Sales collection - SIMPLIFIED for data retrieval
    match /sales/{saleId} {
      allow read: if isAuthenticated() && isUserApproved();
      allow create: if isAdmin() || isSalesPersonnel();
      allow update, delete: if isAdmin();
    }
    
    // Expenses collection
    match /expenses/{expenseId} {
      allow read, write: if isAdmin();
    }
    
    // Approval requests
    match /approvalRequests/{requestId} {
      allow read, write: if isAdmin();
      allow create: if isAuthenticated();
      allow read: if isAuthenticated() && request.auth.uid == resource.data.userId;
    }
    
    // Stock requests
    match /stockRequests/{requestId} {
      allow read, create: if isAuthenticated() && isUserApproved();
      allow update, delete: if isAdmin();
    }
    
    // Other collections - allow basic read for authenticated users
    match /{document=**} {
      allow read, write: if false;
    }
  }
}
